DEF ExampleProgram()
  ; Bu bir örnek KRL programıdır

  DECL INT iCounter
  DECL REAL fSpeed = 0.5
  DECL BOOL bFlag

  ; Başlangıç pozisyonuna git
  PTP HOME

  FOR iCounter = 1 TO 5
    IF bFlag == FALSE THEN
      LIN {X 500, Y 0, Z 600, A 0, B 90, C 0}
    ELSE
      LIN_REL {Z -50}
    ENDIF

    WAIT SEC 1.5
    bFlag = NOT bFlag
  ENDFOR

  ; Fonksiyon çağrısı
  MyCustomFunction(fSpeed)

END

; Değer döndüren fonksiyon
DEF REAL calculate_distance(REAL x1, REAL y1, REAL x2, REAL y2)
    DECL REAL result
    result = SQRT((x2-x1)*(x2-x1) + (y2-y1)*(y2-y1))
    RETURN result
ENDDEF

; Boolean döndüren fonksiyon
DEF BOOL is_position_safe(POS position)
    IF position.X > 1000 OR position.Y > 800 THEN
        RETURN FALSE
    ENDIF
    RETURN TRUE
ENDDEF

; Koşullu return kullanımı
DEF INT check_input(INT value)
    IF value < 0 THEN
        RETURN -1  ; Hata kodu
    ENDIF
    
    IF value > 100 THEN
        RETURN 1   ; Uyarı kodu
    ENDIF
    
    RETURN 0       ; Başarılı
ENDDEF
